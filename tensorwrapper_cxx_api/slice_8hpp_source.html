46 inline bool is_empty_slice_(const sparse_map::Index& lo,
47  const sparse_map::Index& hi) {
48  const auto rank = lo.size();
49 
50  // lo and hi must have same rank
51  TA_ASSERT(rank == hi.size());
52  // Make sure hi is greater than equal to lo, count modes that are equal
53  std::size_t nequal = 0;
54  for(const auto& [l, h] : utilities::Zip(lo, hi)) {
55  TA_ASSERT(h >= l);
56  if(h == l) ++nequal;
57  }
58 
59  // The slice is empty if all modes are equal
60  const bool is_empty = nequal == rank;
61 
62  // Either all of or none of the modes are allowed to be equal. Anything in
63  // between reduces the rank, which we're not supporting at the moment.
64  TA_ASSERT(nequal == 0 || is_empty);
65 
66  return is_empty;
67 }

87 inline auto get_slice_tile_indices_(const TA::TiledRange& tr,
88  const sparse_map::Index& lo,
89  const sparse_map::Index& hi_m1) {
90  const auto temp = get_block_idx(tr, lo);
91  const auto thi_m1 = get_block_idx(tr, hi_m1);
92 
93  // This is the first tile we don't want
94  sparse_map::Index tol(temp.begin(), temp.end());
95  sparse_map::Index thi(thi_m1.begin(), thi_m1.end());
96  for(auto& x : thi.m_index) ++x;
97  return std::make_pair(tol, thi);
98 }

119 template<typename TensorType>
120 auto slice(const TensorType& t, const sparse_map::Index& lo,
121  const sparse_map::Index& hi) {
122  const auto& tr = t.trange();
123  const auto rank = tr.rank();
124  const auto annotation = TA::detail::dummy_annotation(rank);
125 
126  // We rely on error checking done in is_empty_slice_ for everything except
127  // making sure lo and hi are the same rank as the tensor, and that lo and
128  // hi are in the tensor
129  TA_ASSERT(rank == lo.size());
130  const bool is_empty = detail_::is_empty_slice_(lo, hi);
131  TA_ASSERT(!is_empty); // TODO: This function should handle empty slices
132  TA_ASSERT(tr.elements_range().includes(lo));
133 
134  // hi is just past the last element we want, hi "minus 1" is the last
135  // element we want.
136  sparse_map::Index hi_m1(hi);
137  for(auto& x : hi_m1) x -= 1;
138  TA_ASSERT(tr.elements_range().includes(hi_m1));
139 
140  // Now we need to determine if lo/hi are the leading/ending elements of
141  // their respective tiles
142  const bool lo_good = is_tile_lower_bound(tr, lo);
143  const bool hi_good = is_tile_upper_bound(tr, hi);
144 
145  // If we don't have to retile this is easy, just return the slice
146  if(lo_good && hi_good) {
147  auto [tlo, thi] = detail_::get_slice_tile_indices_(tr, lo, hi_m1);
148  TensorType rv;
149  rv(annotation) = t(annotation).block(tlo, thi);
150  return rv;
151  }
152 
153  // Getting here means we need to retile before grabbing the block
154  auto new_tr = insert_tile_boundaries(tr, lo, hi);
155  const TensorType new_t = TA::retile(t, new_tr);
156  auto [tlo, thi] = detail_::get_slice_tile_indices_(new_tr, lo, hi_m1);
157  TensorType rv;
158  rv(annotation) = new_t(annotation).block(tlo, thi);
159  return rv;
160 }
