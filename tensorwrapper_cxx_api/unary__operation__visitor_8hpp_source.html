73 template<typename FloatType>
74 auto make_this_eigen_tensor_() {
75 if(m_pthis_buffer_->size() != m_this_shape_.size()) {
76 std::vector<FloatType> temp_buffer(m_this_shape_.size());
77 *m_pthis_buffer_ = buffer_type(std::move(temp_buffer));
78 }
79 auto this_span =
80 wtf::buffer::contiguous_buffer_cast<FloatType>(*m_pthis_buffer_);
81 return backends::eigen::make_eigen_tensor(this_span, m_this_shape_);
82 }

85 auto make_other_eigen_tensor_(std::span<FloatType> data) {
88
89 using clean_type = std::decay_t<FloatType>;
90 auto* pdata = const_cast<clean_type*>(data.data());
91 std::span<clean_type> non_const_data(pdata, data.size());
92 return backends::eigen::make_eigen_tensor(non_const_data,
93 m_other_shape_);
94 }

96private:

98 label_type m_this_labels_;
99 shape_type m_this_shape_;

146class ApproximatelyEqualVisitor {
147public:
148 explicit ApproximatelyEqualVisitor(double tol) : m_tol_(tol) {}
149
150 template<typename FloatType>
151 bool operator()(const std::span<FloatType> result) {
152 const FloatType zero{0.0};
153 const FloatType ptol = static_cast<FloatType>(m_tol_);
154 for(std::size_t i = 0; i < result.size(); ++i) {
155 auto diff = result[i];
156 if(diff < zero) diff *= -1.0;
157 if(diff >= ptol) return false;
158 }
159 return true;
160 }
161
162private:
163 double m_tol_;
164};

166struct InfinityNormVisitor {
167 template<typename FloatType>
168 auto operator()(const std::span<FloatType> buffer) {
169 std::decay_t<FloatType> max_element{0.0};
170 for(std::size_t i = 0; i < buffer.size(); ++i) {
171 auto elem = types::fabs(buffer[i]);
172 if(elem > max_element) max_element = elem;
173 }
174 return wtf::fp::make_float(max_element);
175 }
