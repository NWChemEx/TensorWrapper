260 offset_vector permutation(const DummyIndices& other) const {
261 if(this->size() != other.size())
262 throw std::runtime_error("Must have same number of dummy indices.");
263
264 if(has_repeated_indices() || other.has_repeated_indices())
265 throw std::runtime_error("Must contain unique dummy indices.");
266
267 offset_vector rv;
268 for(const auto& index : *this) {
269 auto indices = other.find(index);
270 if(indices.empty())
271 throw std::runtime_error("Dummy index not found in other");
272 rv.push_back(indices[0]);
273 }
274 return rv;

501template<typename StringType>
502bool DummyIndices<StringType>::is_hadamard_product(
503 const DummyIndices& lhs, const DummyIndices& rhs) const noexcept {
504 if(has_repeated_indices()) return false;
505 if(lhs.has_repeated_indices()) return false;
506 if(rhs.has_repeated_indices()) return false;
507 if(!is_permutation(lhs)) return false;
508 if(!is_permutation(rhs)) return false;
509 return true;
510}

512template<typename StringType>
513bool DummyIndices<StringType>::is_contraction(
514 const DummyIndices& lhs, const DummyIndices& rhs) const noexcept {
515 if(has_repeated_indices()) return false;
516 if(lhs.has_repeated_indices()) return false;
517 if(rhs.has_repeated_indices()) return false;
518 auto lhs_cap_rhs = lhs.intersection(rhs);
519 if(lhs_cap_rhs.empty()) return false; // No common indices
520 if(!intersection(lhs_cap_rhs).empty())
521 return false; // Common index not summed
522 return true;
523}
