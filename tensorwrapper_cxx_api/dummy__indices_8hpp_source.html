260  offset_vector permutation(const DummyIndices& other) const {
261  if(this->size() != other.size())
262  throw std::runtime_error("Must have same number of dummy indices.");
263 
264  if(has_repeated_indices() || other.has_repeated_indices())
265  throw std::runtime_error("Must contain unique dummy indices.");

512 template<typename StringType>
513 bool DummyIndices<StringType>::is_contraction(
514  const DummyIndices& lhs, const DummyIndices& rhs) const noexcept {
515  if(has_repeated_indices()) return false;
516  if(lhs.has_repeated_indices()) return false;
517  if(rhs.has_repeated_indices()) return false;
518  auto lhs_cap_rhs = lhs.intersection(rhs);
519  if(lhs_cap_rhs.empty()) return false; // No common indices
520  if(!intersection(lhs_cap_rhs).empty())
521  return false; // Common index not summed
522  return true;
523 }
