235  offset_vector permutation(const DummyIndices& other) const {
236  if(this->size() != other.size())
237  throw std::runtime_error("Must have same number of dummy indices.");
238 
239  if(has_repeated_indices() || other.has_repeated_indices())

485 template<typename StringType>
486 bool DummyIndices<StringType>::is_contraction(
487  const DummyIndices& lhs, const DummyIndices& rhs) const noexcept {
488  if(has_repeated_indices()) return false;
489  if(lhs.has_repeated_indices()) return false;
490  if(rhs.has_repeated_indices()) return false;
491  auto lhs_cap_rhs = lhs.intersection(rhs);
492  if(lhs_cap_rhs.empty()) return false; // No common indices
493  if(!intersection(lhs_cap_rhs).empty())
494  return false; // Common index not summed
495  return true;
496 }
