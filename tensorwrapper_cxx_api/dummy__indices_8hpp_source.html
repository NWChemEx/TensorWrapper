166  offset_vector permutation(const DummyIndices& other) const {
167  if(this->size() != other.size())
168  throw std::runtime_error("Must have same number of dummy indices.");
169 
170  if(has_repeated_indices() || other.has_repeated_indices())
171  throw std::runtime_error("Must contain unique dummy indices.");
172 
173  offset_vector rv;
174  for(const auto& index : *this) {
175  auto indices = other.find(index);
176  if(indices.empty())
177  throw std::runtime_error("Dummy index not found in other");
178  rv.push_back(indices[0]);
179  }
180  return rv;
181  }
