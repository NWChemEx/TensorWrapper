103 template<typename TensorType>
104 auto grab_diagonal(TensorType&& t) {
105  auto trange1 = t.trange().dim(0);
106  if(trange1 != t.trange().dim(1))
107  throw std::runtime_error("Expected a square tiling");
108  TA::TiledRange trange{{trange1}};
109  using tensor_type = std::decay_t<TensorType>;
110  using tile_type = typename tensor_type::value_type;
111  t.world().gop.fence();
112  auto l = [=](tile_type& tile, const TA::Range& r) {
113  auto idx = get_block_idx(trange, r);
114  auto t_tile = t.find({idx[0], idx[0]}).get();
115  tile = tile_type(r);
116  for(auto i : r) {
117  std::vector ii{i[0], i[0]};
118  tile[i] = t_tile[ii];
119  }
120  return tile.norm();
121  };
122  auto rv = TA::make_array<tensor_type>(t.world(), trange, l);
123  t.world().gop.fence();
124  return rv;
125 }

138 template<typename T>
139 auto array_from_vec(const std::vector<T>& vec, const TA::TiledRange1& trange,
140  TA::World& world) {
141  using tensor_type = TA::DistArray<TA::Tensor<T>, TA::SparsePolicy>;
142  using tile_type = typename tensor_type::value_type;
143 
144  tensor_type rv(world, TA::TiledRange{trange});
145 
146  for(auto itr = rv.begin(); itr != rv.end(); ++itr) {
147  auto range = rv.trange().make_tile_range(itr.index());
148  tile_type tile(range);
149  for(auto idx : range) { tile(idx) = vec.at(idx[0]); }
150  *itr = tile;
151  }
152  return rv;
153 }

230 template<typename TileType, typename PolicyType, typename AddOp,
231  typename TimesOp, typename ResultType>
232 auto reduce_elementwise(const TA::DistArray<TileType, PolicyType>& lhs,
233  const TA::DistArray<TileType, PolicyType>& rhs,
234  AddOp&& add_op, TimesOp&& times_op, ResultType&& init,
235  std::size_t inner_rank = 0) {
236  using tensor_type = TA::DistArray<TileType, PolicyType>;
237  using add_type = std::decay_t<AddOp>;
238  using times_type = std::decay_t<TimesOp>;
239 
240  detail_::Reducer<tensor_type, add_type, times_type> r(
241  std::forward<AddOp>(add_op), std::forward<TimesOp>(times_op), init);
242 
243  const auto idx =
244  TA::detail::dummy_annotation(lhs.tiles_range().rank(), inner_rank);
245 
246  return lhs(idx).reduce(rhs(idx), std::move(r));
247 }

249 template<typename TileType, typename PolicyType, typename AddOp,
250  typename TimesOp, typename ResultType>
251 auto reduce_tot_elementwise(const TA::DistArray<TileType, PolicyType>& lhs,
252  const TA::DistArray<TileType, PolicyType>& rhs,
253  AddOp&& add_op, TimesOp&& times_op,
254  ResultType&& init) {
255  using inner_type = typename TileType::value_type;
256  using scalar_type = typename TileType::scalar_type;
257  auto inner_times = [times_op{std::forward<TimesOp>(times_op)}](
258  ResultType& result, const scalar_type& first,
259  const scalar_type& second) {
260  result = times_op(first, second);
261  };
262  auto outer_times =
263  [=, inner_times{std::move(inner_times)}](inner_type lhs, inner_type rhs) {
264  return lhs.reduce(rhs, inner_times, add_op, init);
265  };
266  const auto& tile0 = lhs.begin()->get();
267  auto ir = tile0[0].range().rank();
268  return reduce_elementwise(lhs, rhs, std::forward<AddOp>(add_op),
269  outer_times, init, ir)
270  .get();
271 }

319 template<typename T, typename U,
320  typename V = typename std::decay_t<T>::scalar_type>
321 bool allclose(T&& actual, U&& ref, const bool abs_comp = false,
322  V&& rtol = 1.0E-5, V&& atol = 1.0E-8,
323  std::size_t inner_rank = 0) {
324  using tensor_type = std::decay_t<T>;
325  using tile_type = typename tensor_type::value_type;
326  using scalar_type = std::decay_t<V>;
327  static_assert(std::is_same_v<tensor_type, std::decay_t<U>>,
328  "different tensor types is currently unsupported");
329 
330  constexpr bool is_tot = TA::detail::is_tensor_of_tensor_v<tile_type>;
331 
332  // Get a dummy string label (something like "i0, i1, i2, ...")
333  auto idx =
334  TA::detail::dummy_annotation(actual.tiles_range().rank(), inner_rank);
335 
336  // Compute A - B, call result AmB
337  tensor_type AmB;
338  if(abs_comp) {
339  // TODO: There probably is some way to do this without forming the
340  // intermediate tensors
341  auto abs_lambda = [](const scalar_type& val) { return std::fabs(val); };
342  auto abs_actual = apply_elementwise(actual, abs_lambda);
343  auto abs_ref = apply_elementwise(ref, abs_lambda);
344  AmB(idx) = abs_actual(idx) - abs_ref(idx);
345  } else {
346  AmB(idx) = actual(idx) - ref(idx);
347  }
348 
349  auto times_op = [=](scalar_type lhs, scalar_type rhs) {
350  return std::fabs(lhs) <= atol + rtol * std::fabs(rhs);
351  };
352 
353  std::logical_and<bool> add_op;
354 
355  if constexpr(!is_tot) {
356  return reduce_elementwise(AmB, ref, add_op, times_op, true).get();
357  } else {
358  using inner_type = typename tile_type::value_type;
359  auto inner_times = [=](bool& result, const scalar_type& first,
360  const scalar_type& second) {
361  result = times_op(first, second);
362  };
363  auto outer_times = [=](inner_type lhs, inner_type rhs) {
364  return lhs.reduce(rhs, inner_times, add_op, true);
365  };
366  return reduce_elementwise(AmB, ref, add_op, outer_times, true,
367  inner_rank)
368  .get();
369  }
370 }

396 inline TA::TiledRange1 make_1D_trange(std::size_t length, std::size_t tilesize,
397  std::size_t init_offset = 0) {
398  std::vector<std::size_t> bounds;
399  for(std::size_t i = init_offset; i < length; i += tilesize)
400  bounds.push_back(i);
401  bounds.push_back(length);
402  return TA::TiledRange1{bounds.begin(), bounds.end()};

422 template<typename LHSTileType, typename LHSPolicyType, typename RHSTileType,
423  typename RHSPolicyType>
424 bool operator==(const TA::DistArray<LHSTileType, LHSPolicyType>& lhs,
425  const TA::DistArray<RHSTileType, RHSPolicyType>& rhs) {
426  if constexpr(!std::is_same_v<decltype(lhs), decltype(rhs)>)
427  return false;
428  else {
429  if(lhs.is_initialized() != rhs.is_initialized()) return false;
430  if(!lhs.is_initialized()) return true;
431  if(lhs.trange() != rhs.trange()) return false;
432 
433  using scalar_type = typename LHSTileType::scalar_type;
434  using namespace tensorwrapper::ta_helpers;
435  std::logical_and<bool> add_op;
436  std::equal_to<scalar_type> times_op;
437  if constexpr(!TA::detail::is_tensor_of_tensor_v<LHSTileType>) {
438  return reduce_elementwise(lhs, rhs, add_op, times_op, true);
439  } else {
440  return reduce_tot_elementwise(lhs, rhs, add_op, times_op, true);
441  }
442  }
443 }
