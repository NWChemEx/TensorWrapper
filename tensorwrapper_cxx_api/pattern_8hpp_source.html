243template<typename BeginItr, typename EndItr>
244inline auto Pattern::slice(BeginItr first_elem_begin, BeginItr first_elem_end,
245 EndItr last_elem_begin, EndItr last_elem_end) const
246 -> slice_type {
247 auto first_done = [&]() { return first_elem_begin == first_elem_end; };
248 auto last_done = [&]() { return last_elem_begin == last_elem_end; };
249
250 if(first_done() && last_done()) {
251 if(rank() == 0) return slice_type{};
252 throw std::runtime_error("Offset ranks does not match tensor rank");
253 } else if(first_done() || last_done()) {
254 throw std::runtime_error("Offsets do not have the same rank");
255 }
256
257 rank_type counter = 0;
258 for(; !first_done(); ++first_elem_begin, ++last_elem_begin) {
259 if(last_done())
260 throw std::runtime_error("Offsets do not have the same rank.");
261
262 auto fi = *first_elem_begin;
263 auto li = *last_elem_begin;
264 if(li <= fi)
265 throw std::runtime_error("First element in slice must be strictly "
266 "less than last element.");
267
268 ++counter;
269 }
270 if(!last_done())
271 throw std::runtime_error("Offsets do not have the same rank");
272 if(counter != rank())
273 throw std::runtime_error("Offset ranks do not match tensor rank");
274
275 return slice_type(rank());
276}
