418 template<typename IndexType>
419 typename DOMAINPIMPL::value_type DOMAINPIMPL::result_index(
420  const value_type& old) const {
421  if(size() == 0 || old.size() != rank())
422  throw std::out_of_range("Index is not in domain");
423  std::vector<size_type> rv(rank(), 0);
424  for(std::size_t i = 0; i < rank(); ++i) {
425  const auto& offsets = m_mode_map_[i];
426  const auto value = old[i];
427  auto itr = std::find(offsets.begin(), offsets.end(), value);
428  if(itr == offsets.end())
429  throw std::out_of_range("Index is not in domain");
430  rv[i] = std::distance(offsets.begin(), itr);
431  }
432  return IndexType(rv.begin(), rv.end());
433 }

487  const auto& other_mm = other.m_mode_map_;
488  new_modes.insert(new_modes.end(), other_mm.begin(), other_mm.end());

490  std::vector<size_type> new_idx(rank() + other.rank());
491  for(const auto& x : m_domain_) {
492  for(const auto& [i, xi] : utilities::Enumerate(x)) new_idx[i] = xi;
493  for(const auto& y : other.m_domain_) {

495  new_idx[i + rank()] = yi;
