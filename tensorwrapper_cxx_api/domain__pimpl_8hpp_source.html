363 typename DomainPIMPL::value_type DomainPIMPL::result_index(
364  const value_type& old) const {
365  if(size() == 0 || old.size() != rank())
366  throw std::out_of_range("Index is not in domain");
367  std::vector<size_type> rv(rank(), 0);
368  for(std::size_t i = 0; i < rank(); ++i) {
369  const auto& offsets = m_mode_map_[i];
370  const auto value = old[i];
371  auto itr = std::find(offsets.begin(), offsets.end(), value);
372  if(itr == offsets.end())
373  throw std::out_of_range("Index is not in domain");
374  rv[i] = std::distance(offsets.begin(), itr);
375  }
376  return Index(rv.begin(), rv.end());
