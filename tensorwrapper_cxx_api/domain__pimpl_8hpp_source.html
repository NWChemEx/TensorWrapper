379 inline typename DomainPIMPL::value_type DomainPIMPL::result_index(
380  const value_type& old) const {
381  if(size() == 0 || old.size() != rank())
382  throw std::out_of_range("Index is not in domain");
383  std::vector<size_type> rv(rank(), 0);
384  for(std::size_t i = 0; i < rank(); ++i) {
385  const auto& offsets = m_mode_map_[i];
386  const auto value = old[i];
387  auto itr = std::find(offsets.begin(), offsets.end(), value);
388  if(itr == offsets.end())
389  throw std::out_of_range("Index is not in domain");
390  rv[i] = std::distance(offsets.begin(), itr);
391  }
392  return Index(rv.begin(), rv.end());
393 }

419 inline DomainPIMPL& DomainPIMPL::operator*=(const DomainPIMPL& other) {
420  const bool is_empty = m_domain_.empty() || other.m_domain_.empty();
421 
422  if(is_empty) {
423  m_domain_.clear();
424  m_mode_map_.clear();
425  return *this;
426  }
427 
428  boost::container::flat_set<value_type> buffer;
429  std::vector<std::set<size_type>> new_modes(m_mode_map_);
430  const auto& other_mm = other.m_mode_map_;
431  new_modes.insert(new_modes.end(), other_mm.begin(), other_mm.end());
432 
433  std::vector<size_type> new_idx(rank() + other.rank());
434  for(const auto& x : m_domain_) {
435  for(const auto& [i, xi] : utilities::Enumerate(x)) new_idx[i] = xi;
436  for(const auto& y : other.m_domain_) {
437  for(const auto& [i, yi] : utilities::Enumerate(y))
438  new_idx[i + rank()] = yi;
439  buffer.insert(value_type(new_idx));
440  }
441  }
442  m_domain_.swap(buffer);
443  m_mode_map_.swap(new_modes);
444  return *this;
445 }
