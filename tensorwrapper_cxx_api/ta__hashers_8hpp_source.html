102 template<typename TensorType, typename PolicyType>
103 tensorwrapper::detail_::HashValue get_tile_hash_sum(
104  const TA::DistArray<TensorType, PolicyType>& A) {
105  auto& madworld = A.world();
106  auto& mpiworld = madworld.mpi.Get_mpi_comm();
107 
108  // Note: Without the fence orbital space hash tests hang on parallel runs.
109  madworld.gop.fence();
110  tensorwrapper::detail_::HashValue myhash;
111  tensorwrapper::detail_::HashValue mytotal(16, 0);
112  tensorwrapper::detail_::HashValue hashsum(16, 0);
113  // tensorwrapper::detail_::HashType::Hash128 has 16 uint8_t
114 
115  for(auto it = A.begin(); it != A.end(); ++it) {
116  auto tile = A.find(it.index()).get();
117  myhash = tensorwrapper::detail_::make_hash(
118  tensorwrapper::detail_::HashType::Hash128, tile);
119  for(auto i = 0; i < myhash.size(); i++) { mytotal[i] += myhash[i]; }
120  }
121  if(madworld.size() > 1 && !A.pmap().get()->is_replicated()) {
122  // Note: Using &mytotal, &hashsum gives Seg Fault 11.
123  madworld.mpi.Allreduce(mytotal.data(), hashsum.data(), mytotal.size(),
124  MPI_UINT8_T, MPI_SUM);
125  } else {
126  hashsum = mytotal;
127  }
128  madworld.gop.fence();
129  return hashsum;
130 }
