24 inline auto add_tiled_ranges(std::map<types::index, TA::TiledRange1> ranges,
25  const types::index_set& vars,
26  const TA::TiledRange& trange) {
27  for(std::size_t i = 0; i < vars.size(); ++i)
28  ranges[vars[i]] = trange.dim(i);
29  return ranges;
30 }

32 template<typename TileType, typename PolicyType>
33 auto einsum(const types::index& result_idx, const types::index& lhs_idx,
34  const types::index& rhs_idx,
35  const TA::DistArray<TileType, PolicyType>& lhs,
36  const TA::DistArray<TileType, PolicyType>& rhs) {
37  // Figure out what sort of operation the user wants
38  IndexMap im(result_idx, lhs_idx, rhs_idx);
39 
40  // Figure out the TiledRange1 instances for each index
41  std::map<types::index, TA::TiledRange1> ranges;
42  ranges = add_tiled_ranges(std::move(ranges), im.lhs_vars(), lhs.trange());
43  ranges = add_tiled_ranges(std::move(ranges), im.rhs_vars(), rhs.trange());
44 
45  // This lambda is the block kernel for the tensor kernel (there's probably a
46  // way to pass block_kernel directly, but I gave up...)
47  auto l = [](const IndexMap& indices, const types::assoc_range& ranges,
48  const TileType& lhs, const TileType& rhs) {
49  return block_kernel(indices, ranges, lhs, rhs);
50  };
51 
52  // Call the tensor kernel and return the result
53  return tensor_kernel(im, ranges, lhs, rhs, l);
54 }
