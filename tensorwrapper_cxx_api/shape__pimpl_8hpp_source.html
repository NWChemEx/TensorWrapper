262 template<typename FieldType>
263 typename SHAPE_PIMPL::pimpl_pointer SHAPE_PIMPL::slice_(
264  const index_type& _lo, const index_type& _hi) const {
265  if(_lo.size() != m_extents_.size())
266  throw std::runtime_error("Lo bounds do not match extents");
267  if(_hi.size() != m_extents_.size())
268  throw std::runtime_error("Hi bounds do not match extents");
269 
270  tiling_type new_tiling(m_extents_.size());
271  for(auto i = 0ul; i < m_extents_.size(); ++i) {
272  if(_lo[i] < 0 or _lo[i] >= m_extents_[i])
273  throw std::runtime_error("Invalid lo bound");
274  if(_hi[i] > m_extents_[i]) throw std::runtime_error("Invalid hi bound");
275  if(_lo[i] > _hi[i])
276  throw std::runtime_error("Lo must be smaller than Hi");
277 
278  // Shift tiling bounds so the lower bound of the slice is zero.
279  // Push back shifted bounds that are greater than zero,
280  // until we get to one that is greater than or equal to the new
281  // extent of this mode. Then tack the new extent on the end.
282  new_tiling[i] = {0};
283  for(auto bound : m_tiling_[i]) {
284  if(bound < _lo[i]) continue;
285  auto shifted_bound = bound - _lo[i];
286  if(shifted_bound >= (_hi[i] - _lo[i])) break;
287  if(shifted_bound > 0) new_tiling[i].push_back(shifted_bound);
288  }
289  new_tiling[i].push_back((_hi[i] - _lo[i]));
290  }
291 
292  inner_extents_type new_inner_extents;
293  if constexpr(field::is_scalar_field_v<FieldType>) new_inner_extents = 1;
294  if constexpr(field::is_tensor_field_v<FieldType>) {
296  size_type volume = 1;
297  for(auto i = 0ul; i < _lo.size(); ++i) volume *= _hi[i] - _lo[i];
298  for(auto ordinal = 0ul; ordinal < volume; ++ordinal) {
300  std::vector<size_type> temp(_lo.size());
301 
303  auto ord = ordinal;
304  for(auto i = 0ul; i < _lo.size() - 1; ++i) {
305  const auto d = _hi[i] - _lo[i];
306  temp[i] = ord % d;
307  ord = ord / d;
308  }
309  temp[_lo.size() - 1] = ord;
310  index_type new_idx{temp};
311 
313  for(auto i = 0ul; i < _lo.size(); ++i) { temp[i] += _lo[i]; }
314  index_type old_idx{temp};
315 
317  new_inner_extents[new_idx] = m_inner_extents_.at(old_idx);
318  }
319  }
320 
321  return pimpl_pointer(new my_type(new_tiling, new_inner_extents));
322 }
