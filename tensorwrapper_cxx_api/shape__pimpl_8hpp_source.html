209 template<typename FieldType>
210 typename SHAPE_PIMPL::pimpl_pointer SHAPE_PIMPL::slice_(
211  const index_type& _lo, const index_type& _hi) const {
212  if(_lo.size() != m_extents_.size())
213  throw std::runtime_error("Lo bounds do not match extents");
214  if(_hi.size() != m_extents_.size())
215  throw std::runtime_error("Hi bounds do not match extents");
216 
217  extents_type new_extents(m_extents_.size());
218  for(auto i = 0ul; i < m_extents_.size(); ++i) {
219  if(_lo[i] < 0 or _lo[i] >= m_extents_[i])
220  throw std::runtime_error("Invalid lo bound");
221  if(_hi[i] > m_extents_[i]) throw std::runtime_error("Invalid hi bound");
222  if(_lo[i] > _hi[i])
223  throw std::runtime_error("Lo must be smaller than Hi");
224  new_extents[i] = _hi[i] - _lo[i];
225  }
226  return pimpl_pointer(new my_type(new_extents, m_inner_extents_));
227 }
