101  using size_type = typename std::initializer_list<T>::size_type;
102  using extents_type = std::vector<size_type>;
103 
104  if constexpr(N == 0) {
105  return extents_type{};
106  } else {
107  // Handle empty ils
108  if(il.begin() == il.end()) return extents_type(N, 0);
109 
110  const auto& i0 = *(il.begin()); // 0-th element
111 
112  auto sub_extents = il2extents<T, N - 1>(i0);
113  for(const auto& x : il)
114  if(sub_extents != il2extents<T, N - 1>(x))
115  throw std::runtime_error(
116  "Initializer list must be rectangular");

120  }
