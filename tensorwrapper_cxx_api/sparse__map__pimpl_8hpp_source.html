450 template<typename IndIndex, typename DepIndex>
451 SMPIMPL& SMPIMPL::dp_assign_(const my_type& rhs) {
452  if(m_sm_.empty() || rhs.m_sm_.empty()) {
453  m_sm_.clear();
454  return *this;
455  }
456  map_type new_map;
457 
458  using vector_type = std::vector<size_type>;
459  auto new_rank = ind_rank() + rhs.ind_rank();
460  for(auto [lkey, lval] : m_sm_) {
461  for(const auto& [rkey, rval] : rhs.m_sm_) {
462  vector_type new_index;
463  new_index.reserve(new_rank);
464  new_index.insert(new_index.end(), lkey.begin(), lkey.end());
465  new_index.insert(new_index.end(), rkey.begin(), rkey.end());
466  auto new_domain = lval * rval;
467  if(new_domain.empty()) continue;
468  key_type key(new_index);
469  new_map.emplace(std::move(key), std::move(new_domain));
470  }
471  }
472  m_sm_.swap(new_map);
473  return *this;
474 }

476 template<typename IndIndex, typename DepIndex>
477 SMPIMPL& SMPIMPL::prod_assign_(const my_type& rhs) {
478  if(m_sm_.empty())
479  return *this;
480  else if(rhs.m_sm_.empty()) {
481  m_sm_.clear();
482  return *this;
483  }
484  if(ind_rank() != rhs.ind_rank())
485  throw std::runtime_error("Independent ranks do not match");
486 
487  map_type new_map;
488  for(const auto& [lind, ldom] : m_sm_) {
489  if(rhs.count(lind)) {
490  auto new_dom = ldom * rhs.at(lind);
491  if(new_dom.empty()) continue;
492  new_map.emplace(lind, std::move(new_dom));
493  }
494  }
495  m_sm_.swap(new_map);
496  return *this;
497 }

501  if(rhs.m_sm_.empty())
502  return *this;

504  m_sm_ = rhs.m_sm_;
505  return *this;

517 template<typename IndIndex, typename DepIndex>
518 SMPIMPL& SMPIMPL::int_assign_(const my_type& rhs) {
519  if(m_sm_.empty())
520  return *this;
521  else if(rhs.m_sm_.empty() || (ind_rank() != rhs.ind_rank())) {
522  m_sm_.clear();
523  return *this;
524  }
525 
526  map_type new_map;
527  for(const auto& [lind, ldom] : m_sm_) {
528  if(!rhs.count(lind)) continue;
529  const auto intersection = ldom ^ rhs.at(lind);
530  for(const auto& dep : intersection) new_map[lind].insert(dep);
531  }
532  m_sm_.swap(new_map);
533  return *this;
