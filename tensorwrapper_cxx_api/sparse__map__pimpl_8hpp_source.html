401 inline auto& SparseMapPIMPL::direct_product_assign(const SparseMapPIMPL& rhs) {
402  if(m_sm_.empty() || rhs.m_sm_.empty()) {
403  m_sm_.clear();
404  return *this;
405  }
406  map_type new_map;
407 
408  using vector_type = std::vector<size_type>;
409  auto new_rank = ind_rank() + rhs.ind_rank();
410  for(auto [lkey, lval] : m_sm_) {
411  for(const auto& [rkey, rval] : rhs.m_sm_) {
412  vector_type new_index;
413  new_index.reserve(new_rank);
414  new_index.insert(new_index.end(), lkey.begin(), lkey.end());
415  new_index.insert(new_index.end(), rkey.begin(), rkey.end());
416  auto new_domain = lval * rval;
417  if(new_domain.empty()) continue;
418  key_type key(new_index);
419  new_map.emplace(std::move(key), std::move(new_domain));
420  }
421  }
422  m_sm_.swap(new_map);
423  return *this;
424 }

426 inline SparseMapPIMPL& SparseMapPIMPL::operator*=(const SparseMapPIMPL& rhs) {
427  if(m_sm_.empty())
428  return *this;
429  else if(rhs.m_sm_.empty()) {
430  m_sm_.clear();
431  return *this;
432  }
433  if(ind_rank() != rhs.ind_rank())
434  throw std::runtime_error("Independent ranks do not match");
435 
436  map_type new_map;
437  for(const auto& [lind, ldom] : m_sm_) {
438  if(rhs.count(lind)) {
439  auto new_dom = ldom * rhs.at(lind);
440  if(new_dom.empty()) continue;
441  new_map.emplace(lind, std::move(new_dom));
442  }
443  }
444  m_sm_.swap(new_map);
445  return *this;
446 }

465 inline SparseMapPIMPL& SparseMapPIMPL::operator^=(const SparseMapPIMPL& rhs) {
466  if(m_sm_.empty())
467  return *this;
468  else if(rhs.m_sm_.empty() || (ind_rank() != rhs.ind_rank())) {
469  m_sm_.clear();
470  return *this;
471  }
472 
473  map_type new_map;
474  for(const auto& [lind, ldom] : m_sm_) {
475  if(!rhs.count(lind)) continue;
476  const auto intersection = ldom ^ rhs.at(lind);
477  for(const auto& dep : intersection) new_map[lind].insert(dep);
478  }
479  m_sm_.swap(new_map);
480  return *this;
