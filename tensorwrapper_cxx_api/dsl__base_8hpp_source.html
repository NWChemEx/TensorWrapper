265 void assert_indices_match_rank_(const_labeled_reference other) const {
266 const auto rank = other.object().rank();
267 const auto n = other.labels().size();
268 if(rank == n) return;
269 auto msg = std::to_string(n) +
270 " dummy indices is incompatible with an object"
271 " with rank =" +
272 std::to_string(rank);
273 throw std::runtime_error(msg);
274 }
275
277 void assert_is_subset_(const label_type& output,
278 const label_type& input) const {
279 // Subset would have equality
280 if(output.intersection(input).size() < output.unique_index_size()) {
281 throw std::runtime_error(
282 "Output indices must be a subset of input indices");
283 }
284 }
285
287 void assert_is_permutation_(const label_type& lhs,
288 const label_type& rhs) const {
289 if(lhs.is_permutation(rhs)) return;
290 throw std::runtime_error(
291 "Dummy indices are not related via permutation.");
292 }
