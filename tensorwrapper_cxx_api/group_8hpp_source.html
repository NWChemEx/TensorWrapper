425template<typename BeginItr, typename EndItr>
426inline auto Group::slice(BeginItr first_elem_begin, BeginItr first_elem_end,
427 EndItr last_elem_begin, EndItr last_elem_end) const
428 -> slice_type {
429 if(size() != 0)
430 throw std::runtime_error("Slicing of non-trivial symmetry NYI");
431
432 auto first_done = [&]() { return first_elem_begin == first_elem_end; };
433 auto last_done = [&]() { return last_elem_begin == last_elem_end; };
434
435 if(first_done() && last_done()) {
436 if(rank() == 0) return slice_type{};
437 throw std::runtime_error("Offset ranks does not match tensor rank");
438 } else if(first_done() || last_done()) {
439 throw std::runtime_error("Offsets do not have the same rank");
440 }
441
442 rank_type counter = 0;
443 for(; !first_done(); ++first_elem_begin, ++last_elem_begin) {
444 if(last_done())
445 throw std::runtime_error("Offsets do not have the same rank.");
446
447 auto fi = *first_elem_begin;
448 auto li = *last_elem_begin;
449 if(li <= fi)
450 throw std::runtime_error("First element in slice must be strictly "
451 "less than last element.");
452
453 ++counter;
454 }
455 if(!last_done())
456 throw std::runtime_error("Offsets do not have the same rank");
457 if(counter != rank())
458 throw std::runtime_error("Offset ranks do not match tensor rank");
459
460 return slice_type(rank());
461}
