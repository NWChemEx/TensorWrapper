181template<typename Derived>
182template<typename BeginItr, typename EndItr>
183auto SmoothCommon<Derived>::slice(BeginItr first_elem_begin,
184 BeginItr first_elem_end,
185 EndItr last_elem_begin,
186 EndItr last_elem_end) const -> slice_type {
187 std::vector<size_type> new_extents;
188
189 auto first_done = [&]() { return first_elem_begin == first_elem_end; };
190 auto last_done = [&]() { return last_elem_begin == last_elem_end; };
191
192 if(first_done() && last_done()) {
193 // TODO: Assert rank 0
194 return slice_type{};
195 } else if(first_done() || last_done()) {
196 throw std::runtime_error("Ranges were NOT equal");
197 }
198
199 for(; !first_done(); ++first_elem_begin, ++last_elem_begin) {
200 if(last_done()) // Last ended before first
201 throw std::runtime_error("Ranges were NOT equal");
202
203 const auto ei = *last_elem_begin;
204 const auto bi = *first_elem_begin;
205 if(bi >= ei)
206 throw std::runtime_error("begin index must be < end index");
207
208 new_extents.push_back(ei - bi);
209 }
210 if(!last_done()) { throw std::runtime_error("Ranges were NOT equal"); }
211
212 // TODO: assert rank == new_extents.size()
213 return slice_type(new_extents.begin(), new_extents.end());
214}
