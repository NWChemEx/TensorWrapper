46 inline auto uninject_index_(
47  const Index& idx, const std::map<std::size_t, std::size_t>& injections) {
48  const auto r = idx.size();
49  if(injections.empty()) return idx;
50  for(const auto& [k, v] : injections)
51  if(k >= r)
52  throw std::runtime_error("Mode: " + std::to_string(k) +
53  " is not in range [0, " +
54  std::to_string(r) + ")");
55  std::vector<std::size_t> uninjected_idx(r - injections.size());
56  for(std::size_t i = 0, counter = 0; i < r; ++i) {
57  if(!injections.count(i)) {
58  uninjected_idx[counter] = idx[i];
59  ++counter;
60  }
61  }
62  return Index(std::move(uninjected_idx));
63 }

80 template<typename TileType, typename T>
81 auto make_tot_tile_(TileType tile, const SparseMap& sm, const T& tensor,
82  const std::map<std::size_t, std::size_t>& ind2mode = {}) {
83  using inner_tile_t = typename std::decay_t<TileType>::value_type;
84 
85  const auto& trange = tensor.trange(); // Trange of "dense"
86  const auto ind_rank = sm.ind_rank(); // Rank of inner tensor
87  const auto inj_rank = ind_rank + ind2mode.size(); // Rank of injected index
88  const bool do_inj = !ind2mode.empty(); // Are we injecting?
89 
90  // Move allocations out of loops
91  std::map<std::size_t, std::size_t> injections; // Map for injections
92 
93  // Loop over outer-elemental indices
94  for(const auto& oeidx_v : tile.range()) {
95  const Index oeidx(oeidx_v.begin(), oeidx_v.end());
96 
97  // Handle scenario where independent index has no domain
98  if(!sm.count(oeidx)) {
99  tile[oeidx] = inner_tile_t{};
100  continue;
101  }
102 
103  const auto& d = sm.at(oeidx); // The elements in the inner tile
104  inner_tile_t buffer(TA::Range(d.result_extents()), 0.0);
105 
106  // Determine tiles to retrieve using injected domain
107  // TODO: This is just a copy of d if do_inj == false
108  for(const auto& [k, v] : ind2mode) injections[v] = oeidx[k];
109  auto injected_d = d.inject(injections);
110  auto tdomain = detail_::tile_domain(injected_d, trange);
111 
112  for(const auto& itidx : tdomain) { // Loop over inner-tile indices
113  if(tensor.is_zero(itidx)) continue;
114  inner_tile_t t = tensor.find(itidx);
115 
116  // It's not clear to me whether the injection alters the order. If
117  // the indices in injected_d are ordered such that the i-th index
118  // of injected_d is the i-th index of d (with the former containing
119  // the injected modes) then we can loop over injected_d, d zipped
120  // together and avoid the uninjection
121  for(const auto& ieidx : injected_d) { // Loop over inner-element
122  if(t.range().includes(ieidx)) { // Is element in tile?
123  // Remove injected modes
124  auto lhs_idx = uninject_index_(ieidx, injections);
125  // map it to output
126  buffer[d.result_index(lhs_idx)] = t[ieidx];
127  }
128  }
129  }
130  tile[oeidx] = buffer;
131  }
132  return tile;
133 }

158 template<typename T>
159 auto from_sparse_map(const SparseMap& esm, const T& tensor,
160  const TA::TiledRange outer_trange,
161  const std::map<std::size_t, std::size_t>& ind2mode = {}) {
162  using scalar_type = typename T::scalar_type;
163  using tot_type =
164  TA::DistArray<TA::Tensor<TA::Tensor<scalar_type>>, TA::SparsePolicy>;
165 
166  if(esm.dep_rank() + ind2mode.size() != tensor.trange().rank())
167  throw std::runtime_error("Ranks don't work out.");
168 
169  auto tesm = detail_::tile_independent_indices(esm, outer_trange);
170  auto l = [=](auto& tile, const auto& range) {
171  using tile_type = std::decay_t<decltype(tile)>;
172  auto otidx = ta_helpers::get_block_idx(outer_trange, range);
173  if(!tesm.count(Index(otidx.begin(), otidx.end()))) return 0.0;
174  tile = detail_::make_tot_tile_(tile_type(range), esm, tensor, ind2mode);
175  return tile.norm();
176  };
177 
178  return TA::make_array<tot_type>(tensor.world(), outer_trange, l);
179 }
