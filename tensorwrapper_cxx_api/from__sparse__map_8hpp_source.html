29 inline auto uninject_index_(
30  const ElementIndex& idx,
31  const std::map<std::size_t, std::size_t>& injections) {
32  const auto r = idx.size();
33  if(injections.empty()) return idx;
34  for(const auto& [k, v] : injections)
35  if(k >= r)
36  throw std::runtime_error("Mode: " + std::to_string(k) +
37  " is not in range [0, " +
38  std::to_string(r) + ")");
39  std::vector<std::size_t> uninjected_idx(r - injections.size());
40  for(std::size_t i = 0, counter = 0; i < r; ++i) {
41  if(!injections.count(i)) {
42  uninjected_idx[counter] = idx[i];
43  ++counter;
44  }
45  }
46  return ElementIndex(std::move(uninjected_idx));
47 }

65 auto make_tot_tile_(TileType tile,
66  const SparseMap<ElementIndex, ElementIndex>& sm,
67  const T& tensor,
68  const std::map<std::size_t, std::size_t>& ind2mode = {}) {
69  using inner_tile_t = typename std::decay_t<TileType>::value_type;
70 
71  const auto& trange = tensor.trange(); // Trange of "dense"
72  const auto ind_rank = sm.ind_rank(); // Rank of inner tensor
73  const auto inj_rank = ind_rank + ind2mode.size(); // Rank of injected index
74  const bool do_inj = !ind2mode.empty(); // Are we injecting?
75 
76  // Move allocations out of loops
77  std::map<std::size_t, std::size_t> injections; // Map for injections
78 
79  for(const auto& oeidx_v :
80  tile.range()) { // Loop over outer-elemental indices
81  const ElementIndex oeidx(oeidx_v.begin(), oeidx_v.end());
82 
83  // Handle scenario where independent index has no domain
84  if(!sm.count(oeidx)) {
85  tile[oeidx] = inner_tile_t{};
86  continue;
87  }
88 
89  const auto& d = sm.at(oeidx); // The elements in the inner tile
90  inner_tile_t buffer(TA::Range(d.result_extents()), 0.0);
91 
92  // Determine tiles to retrieve using injected domain
93  // TODO: This is just a copy of d if do_inj == false
94  for(const auto& [k, v] : ind2mode) injections[v] = oeidx[k];
95  auto injected_d = d.inject(injections);
96  Domain<TileIndex> tdomain(trange, injected_d);
97 
98  for(const auto& itidx : tdomain) { // Loop over inner-tile indices
99  if(tensor.is_zero(itidx)) continue;
100  inner_tile_t t = tensor.find(itidx);
101 
102  // It's not clear to me whether the injection alters the order. If
103  // the indices in injected_d are ordered such that the i-th index
104  // of injected_d is the i-th index of d (with the former containing
105  // the injected modes) then we can loop over injected_d, d zipped
106  // together and avoid the uninjection
107  for(const auto& ieidx : injected_d) { // Loop over inner-element
108  if(t.range().includes(ieidx)) { // Is element in tile?
109  // Remove injected modes
110  auto lhs_idx = uninject_index_(ieidx, injections);
111  // map it to output
112  buffer[d.result_index(lhs_idx)] = t[ieidx];
113  }
114  }
115  }
116  tile[oeidx] = buffer;
117  }
118  return tile;
119 }
