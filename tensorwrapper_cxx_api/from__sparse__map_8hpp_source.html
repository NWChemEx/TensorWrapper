30 inline auto uninject_index_(
31  const Index& idx, const std::map<std::size_t, std::size_t>& injections) {
32  const auto r = idx.size();
33  if(injections.empty()) return idx;
34  for(const auto& [k, v] : injections)
35  if(k >= r)
36  throw std::runtime_error("Mode: " + std::to_string(k) +
37  " is not in range [0, " +
38  std::to_string(r) + ")");
39  std::vector<std::size_t> uninjected_idx(r - injections.size());
40  for(std::size_t i = 0, counter = 0; i < r; ++i) {
41  if(!injections.count(i)) {
42  uninjected_idx[counter] = idx[i];
43  ++counter;
44  }
45  }
46  return Index(std::move(uninjected_idx));
47 }

64 template<typename TileType, typename T>
65 auto make_tot_tile_(TileType tile, const SparseMap& sm, const T& tensor,
66  const std::map<std::size_t, std::size_t>& ind2mode = {}) {
67  using inner_tile_t = typename std::decay_t<TileType>::value_type;
68 
69  const auto& trange = tensor.trange(); // Trange of "dense"
70  const auto ind_rank = sm.ind_rank(); // Rank of inner tensor
71  const auto inj_rank = ind_rank + ind2mode.size(); // Rank of injected index
72  const bool do_inj = !ind2mode.empty(); // Are we injecting?
73 
74  // Move allocations out of loops
75  std::map<std::size_t, std::size_t> injections; // Map for injections
76 
77  // Loop over outer-elemental indices
78  for(const auto& oeidx_v : tile.range()) {
79  const Index oeidx(oeidx_v.begin(), oeidx_v.end());
80 
81  // Handle scenario where independent index has no domain
82  if(!sm.count(oeidx)) {
83  tile[oeidx] = inner_tile_t{};
84  continue;
85  }
86 
87  const auto& d = sm.at(oeidx); // The elements in the inner tile
88  inner_tile_t buffer(TA::Range(d.result_extents()), 0.0);
89 
90  // Determine tiles to retrieve using injected domain
91  // TODO: This is just a copy of d if do_inj == false
92  for(const auto& [k, v] : ind2mode) injections[v] = oeidx[k];
93  auto injected_d = d.inject(injections);
94  auto tdomain = detail_::tile_domain(injected_d, trange);
95 
96  for(const auto& itidx : tdomain) { // Loop over inner-tile indices
97  if(tensor.is_zero(itidx)) continue;
98  inner_tile_t t = tensor.find(itidx);
99 
100  // It's not clear to me whether the injection alters the order. If
101  // the indices in injected_d are ordered such that the i-th index
102  // of injected_d is the i-th index of d (with the former containing
103  // the injected modes) then we can loop over injected_d, d zipped
104  // together and avoid the uninjection
105  for(const auto& ieidx : injected_d) { // Loop over inner-element
106  if(t.range().includes(ieidx)) { // Is element in tile?
107  // Remove injected modes
108  auto lhs_idx = uninject_index_(ieidx, injections);
109  // map it to output
110  buffer[d.result_index(lhs_idx)] = t[ieidx];
111  }
112  }
113  }
114  tile[oeidx] = buffer;
115  }
116  return tile;
117 }

142 template<typename T>
143 auto from_sparse_map(const SparseMap& esm, const T& tensor,
144  const TA::TiledRange outer_trange,
145  const std::map<std::size_t, std::size_t>& ind2mode = {}) {
146  using scalar_type = typename T::scalar_type;
147  using tot_type =
148  TA::DistArray<TA::Tensor<TA::Tensor<scalar_type>>, TA::SparsePolicy>;
149 
150  if(esm.dep_rank() + ind2mode.size() != tensor.trange().rank())
151  throw std::runtime_error("Ranks don't work out.");
152 
153  auto tesm = detail_::tile_independent_indices(esm, outer_trange);
154  auto l = [=](auto& tile, const auto& range) {
155  using tile_type = std::decay_t<decltype(tile)>;
156  auto otidx = ta_helpers::get_block_idx(outer_trange, range);
157  if(!tesm.count(Index(otidx.begin(), otidx.end()))) return 0.0;
158  tile = detail_::make_tot_tile_(tile_type(range), esm, tensor, ind2mode);
159  return tile.norm();
160  };
161 
162  return TA::make_array<tot_type>(tensor.world(), outer_trange, l);
163 }
