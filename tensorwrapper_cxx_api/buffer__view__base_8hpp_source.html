43template<typename BufferBaseType>
44class BufferViewBase : public BufferBaseCommon<BufferViewBase<BufferBaseType>> {
45private:
46 static_assert(std::is_same_v<BufferBaseType, BufferBase> ||
47 std::is_same_v<BufferBaseType, const BufferBase>,
48 "BufferViewBase BufferBaseType must be BufferBase or "
49 "const BufferBase");

55 using pimpl_type = detail_::BufferViewBasePIMPL<BufferBaseType>;
56 using pimpl_reference = pimpl_type&;
57 using const_pimpl_reference = const pimpl_type&;

59public:
60 using typename my_base_type::const_layout_reference;
61 using typename my_base_type::layout_pointer;

63 using typename my_base_type::layout_type;

81 explicit BufferViewBase(BufferBaseType& buffer) noexcept :

104 BufferViewBase(const BufferViewBase& other) noexcept :

126 BufferViewBase& operator=(const BufferViewBase& rhs) noexcept {

151 bool operator!=(const BufferViewBase& rhs) const noexcept {

155protected:

157 friend class BufferBase;

159 // -------------------------------------------------------------------------
160 // -- BufferBaseCommon hooks
161 // -------------------------------------------------------------------------
162
163 bool has_layout_() const noexcept {
164 return m_pimpl_ != nullptr && m_pimpl_->has_layout();
165 }
166
167 layout_reference layout_() { return pimpl_().layout(); }
168
169 const_layout_reference layout_() const { return pimpl_().layout(); }
170
171 // Will be polymorphic eventually
172 template<typename OtherBufferBaseType>
173 bool approximately_equal_(const BufferViewBase<OtherBufferBaseType>& rhs,
174 double) const {
175 return *this == rhs;
176 }
177
178 // Will be polymorphic eventually
179 bool approximately_equal_(const BufferBase& rhs, double) const {
180 return *this == rhs;
181 }
182
183private:
184 void assert_pimpl_() const {
185 if(!m_pimpl_) {
186 throw std::runtime_error(
187 "BufferViewBase has no PIMPL. Was it default initialized?");
188 }
189 }
190 pimpl_reference pimpl_() {
191 assert_pimpl_();
192 return *m_pimpl_;
193 }
194
195 const_pimpl_reference pimpl_() const {
196 assert_pimpl_();
197 return *m_pimpl_;
198 }
199
201 std::unique_ptr<pimpl_type> m_pimpl_;
202};
