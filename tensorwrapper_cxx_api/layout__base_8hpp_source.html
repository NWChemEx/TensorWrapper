299 inline void LayoutBase::assert_valid_state_() const {
300  if(m_shape_ == nullptr) throw std::runtime_error("Shape can't be null");
301  if(m_symmetry_ == nullptr)
302  throw std::runtime_error("Symmetry can't be null");
303  if(m_sparsity_ == nullptr)
304  throw std::runtime_error("Sparsity can't be null");
305 
306  bool same_rank = m_shape_->rank() == m_symmetry_->rank();
307  same_rank = same_rank && m_shape_->rank() == m_sparsity_->rank();
308  if(!same_rank) throw std::runtime_error("Must have same rank");
309 }
