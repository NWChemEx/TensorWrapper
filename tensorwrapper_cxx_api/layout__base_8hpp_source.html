311inline void LayoutBase::assert_valid_state_() const {
312 if(m_shape_ == nullptr) throw std::runtime_error("Shape can't be null");
313 if(m_symmetry_ == nullptr)
314 throw std::runtime_error("Symmetry can't be null");
315 if(m_sparsity_ == nullptr)
316 throw std::runtime_error("Sparsity can't be null");
317
318 bool same_rank = m_shape_->rank() == m_symmetry_->rank();
319 same_rank = same_rank && m_shape_->rank() == m_sparsity_->rank();
320 if(!same_rank) throw std::runtime_error("Must have same rank");
321}
