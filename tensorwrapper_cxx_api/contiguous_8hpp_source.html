362template<concepts::FloatingPoint T>
363Contiguous make_contiguous(const shape::ShapeBase& shape, T initial_value) {
364 auto smooth_view = shape.as_smooth();
365 using size_type = typename decltype(smooth_view)::size_type;
366 std::vector<size_type> extents(smooth_view.rank());
367 for(size_type i = 0; i < smooth_view.rank(); ++i)
368 extents[i] = smooth_view.extent(i);
369 shape::Smooth smooth_shape(extents.begin(), extents.end());
370 std::vector<T> elements(smooth_view.size(), initial_value);
371 return Contiguous(std::move(elements), std::move(smooth_shape));
372}

374template<concepts::FloatingPoint T>
375Contiguous make_contiguous(const shape::ShapeBase& shape) {
376 return make_contiguous(shape, static_cast<T>(0));
377}
