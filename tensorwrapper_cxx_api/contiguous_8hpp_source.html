364template<concepts::FloatingPoint T>
365Contiguous make_contiguous(const shape::ShapeBase& shape, T initial_value) {
366 auto smooth_view = shape.as_smooth();
367 using size_type = typename decltype(smooth_view)::size_type;
368 std::vector<size_type> extents(smooth_view.rank());
369 for(size_type i = 0; i < smooth_view.rank(); ++i)
370 extents[i] = smooth_view.extent(i);
371 shape::Smooth smooth_shape(extents.begin(), extents.end());
372 std::vector<T> elements(smooth_view.size(), initial_value);
373 return Contiguous(std::move(elements), std::move(smooth_shape));
374}

376template<concepts::FloatingPoint T>
377Contiguous make_contiguous(const shape::ShapeBase& shape) {
378 return make_contiguous(shape, static_cast<T>(0));
379}
