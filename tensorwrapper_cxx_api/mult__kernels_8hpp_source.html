59  if(l_count == 1 && r_count == 1) {
60  const auto& ostr = result.annotation();

62  const auto& rstr = rhs.annotation();
63  auto& lt = lhs.array();

67  return std::make_pair(true, result);
68  }

105  template<typename ResultType, typename LHSType, typename RHSType>
106  static return_type eval(ResultType&& result, LHSType&& lhs, RHSType&& rhs) {
107  auto lhs_variant = lhs.variant(result);
108  auto rhs_variant = rhs.variant(result);
109  auto result_variant = result.variant(result);
110  using result_variant_t = std::decay_t<decltype(result_variant)>;
111  auto l = [&](auto&& result) {
112  auto m = [&](auto&& lhs) {
113  auto n = [&](auto&& rhs) {
114  auto [done, temp] = use_einsum_(result, lhs, rhs);
115  if(done) return result_variant_t(temp);
116  result = lhs * rhs;
117  return result_variant_t(result);
118  };
119  return std::visit(n, rhs_variant);
120  };
121  return std::visit(m, lhs_variant);
122  };
123  return std::visit(l, result_variant);
