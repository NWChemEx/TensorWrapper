44template<typename KernelType, typename BufferType, typename... Args>
45decltype(auto) floating_point_dispatch(KernelType&& kernel, BufferType&& buffer,
46 Args&&... args) {
47 using buffer_clean = std::decay_t<BufferType>;
48 using buffer_base = buffer::BufferBase;
49 constexpr bool is_buffer = std::is_base_of_v<buffer_base, buffer_clean>;
50 static_assert(is_buffer);
51
52 using types::udouble;
53 using types::ufloat;
54
55 if(allocator::Eigen<float>::can_rebind(buffer)) {
56 return kernel.template run<float>(buffer, std::forward<Args>(args)...);
57 } else if(allocator::Eigen<double>::can_rebind(buffer)) {
58 return kernel.template run<double>(buffer, std::forward<Args>(args)...);
59 } else if(allocator::Eigen<ufloat>::can_rebind(buffer)) {
60 return kernel.template run<ufloat>(buffer, std::forward<Args>(args)...);
61 } else if(allocator::Eigen<udouble>::can_rebind(buffer)) {
62 return kernel.template run<udouble>(buffer,
63 std::forward<Args>(args)...);
64 } else {
65 throw std::runtime_error("Can't rebind buffer to Contiguous<>");
66 }
67}
