50 inline auto make_pair_map(const SparseMap& L_ia, const SparseMap& L_ij) {
51  if(L_ia.ind_rank() != L_ij.ind_rank()) {
52  std::string msg = "Independent rank of L_ia (" +
53  std::to_string(L_ia.ind_rank()) + ") does not match" +
54  " the independent rank of L_ij (" +
55  std::to_string(L_ij.ind_rank()) + ").";
56  throw std::runtime_error(msg);
57  }
58 
59  if(L_ia.ind_rank() != L_ij.dep_rank()) {
60  std::string msg = "Independent rank of L_ia (" +
61  std::to_string(L_ia.ind_rank()) + ") does not match" +
62  " the dependent rank of L_ij (" +
63  std::to_string(L_ij.dep_rank()) + ").";
64  throw std::runtime_error(msg);
65  }
66 
67  if(L_ia.ind_rank() != 1)
68  throw std::runtime_error("Can only handle rank 1 independent indices");
69 
70  SparseMap rv;
71 
72  for(const auto& [i, domain_i] : L_ij) {
73  for(const auto& j : domain_i) {
74  Index ij{i[0], j[0]};
75  for(const auto& a : L_ia.at(i)) rv.add_to_domain(ij, a);
76  for(const auto& a : L_ia.at(j)) rv.add_to_domain(ij, a);
77  }
78  }
79  return rv;
80 }
