26 template<typename T, typename FieldType = field::Scalar>
27 auto ta_to_tw(T&& t) {
28  static_assert(field::is_scalar_field_v<FieldType>,
29  "Only scalar fields are presently implemented");
30 
31  if(!t.is_initialized()) return TensorWrapper<FieldType>();
32 
33  // Step 0: Make shape
34  using shape_type = Shape<FieldType>;
35  using tiling_type = typename shape_type::tiling_type;
36  tiling_type tiling(t.trange().rank());
37  for(std::size_t i = 0; i < tiling.size(); ++i) {
38  auto dim_range = t.trange().dim(i);
39  const auto ntiles = std::distance(dim_range.begin(), dim_range.end());
40  tiling[i] = typename tiling_type::value_type(ntiles + 1);
41  std::transform(dim_range.begin(), dim_range.end(), tiling[i].begin(),
42  [](const auto& p) { return p.first; });
43  tiling[i].back() = (dim_range.end() - 1)->second;
44  }
45  auto pshape = std::make_unique<shape_type>(tiling);
46 
47  // Step 1: Make allocator
48  auto palloc = allocator::ta_allocator<FieldType>();
49 
50  // Step 2: TA tensor into a TABufferPIMPL
51  using ta_pimpl_type = buffer::detail_::TABufferPIMPL<FieldType>;
52  auto pt = std::make_unique<ta_pimpl_type>(std::forward<T>(t));
53 
54  // Step 3: Move BufferPIMPL into a Buffer
55  using buffer_type = buffer::Buffer<FieldType>;
56  auto pbuffer = std::make_unique<buffer_type>(std::move(pt));
57 
58  // Step 4: Move buffer, shape, and allocator into TensorWrapperPIMPL
59  using pimpl_type = TensorWrapperPIMPL<FieldType>;
60  auto ppimpl = std::make_unique<pimpl_type>(
61  std::move(pbuffer), std::move(pshape), std::move(palloc));
62 
63  // Finally make the tensor
64  return TensorWrapper<FieldType>(std::move(ppimpl));
65 }
