48 template<typename... Args>
49 auto recurse_for_extent_(const std::string& label,
50  dsl::Labeled<shape::ShapeBase> shape, Args&&... args) {
51  auto idx = shape.labels().find(label);
52  if(idx.empty()) {
53  if constexpr(sizeof...(args) > 0) {
54  return recurse_for_extent_(label, std::forward<Args>(args)...);
55  } else {
56  throw std::runtime_error("Label " + label +
57  " not found in any provided shapes");
58  }
59  } else {
60  return shape.object().as_smooth().extent(idx[0]);
61  }
62 }
