48template<typename... Args>
49auto recurse_for_extent_(const std::string& label,
50 dsl::Labeled<shape::ShapeBase> shape, Args&&... args) {
51 auto idx = shape.labels().find(label);
52 if(idx.empty()) {
53 if constexpr(sizeof...(args) > 0) {
54 return recurse_for_extent_(label, std::forward<Args>(args)...);
55 } else {
56 throw std::runtime_error("Label " + label +
57 " not found in any provided shapes");
58 }
59 } else {
60 return shape.object().as_smooth().extent(idx[0]);
61 }
62}

91template<typename StringType, typename... Args>
92shape::Smooth shape_from_labels(const dsl::DummyIndices<StringType>& labels,
93 Args&&... labeled_shapes) {
94 static_assert(sizeof...(Args) > 0,
95 "Must provide at least one labeled shape");
96
97 std::vector<std::size_t> extents;
98 for(const auto& label : labels) {
99 extents.push_back(recurse_for_extent_(label, labeled_shapes...));
100 }
101
102 return shape::Smooth(extents.begin(), extents.end());
103}
