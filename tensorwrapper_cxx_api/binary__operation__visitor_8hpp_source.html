39class BinaryOperationVisitor : private UnaryOperationVisitor {
40private:

53 BinaryOperationVisitor(buffer_type& this_buffer, label_type this_labels,
54 shape_type this_shape, label_type lhs_labels,
55 shape_type lhs_shape, label_type rhs_labels,
56 shape_type rhs_shape) :
57 UnaryOperationVisitor(this_buffer, this_labels, this_shape, lhs_labels,
58 lhs_shape),
59 m_rhs_labels_(std::move(rhs_labels)),
60 m_rhs_shape_(std::move(rhs_shape)) {}
61
62 using base_class::this_labels;
63 using base_class::this_shape;
64
65 const auto& lhs_shape() const { return other_shape(); }
66 const auto& rhs_shape() const { return m_rhs_shape_; }
67
68 const auto& lhs_labels() const { return other_labels(); }
69 const auto& rhs_labels() const { return m_rhs_labels_; }
70
71 template<typename LHSType, typename RHSType>
72 requires(!std::is_same_v<LHSType, RHSType>)
73 void operator()(std::span<const LHSType>, std::span<const RHSType>) const {
74 throw std::runtime_error(
75 "BinaryOperationVisitor: Mixed types not supported");
76 }
77
78protected:
79 using base_class::make_this_eigen_tensor_;
80
81 template<typename FloatType>
82 auto make_lhs_eigen_tensor_(std::span<FloatType> data) {
83 return base_class::make_other_eigen_tensor_(data);
84 }
85
86 template<typename FloatType>
87 auto make_rhs_eigen_tensor_(std::span<FloatType> data) {
90
91 using clean_type = std::decay_t<FloatType>;
92 auto* pdata = const_cast<clean_type*>(data.data());
93 std::span<clean_type> non_const_data(pdata, data.size());
94 return backends::eigen::make_eigen_tensor(non_const_data, m_rhs_shape_);
95 }
96
97private:
98 label_type m_rhs_labels_;
99 shape_type m_rhs_shape_;
100};

103class AdditionVisitor : public BinaryOperationVisitor {
104public:
105 using BinaryOperationVisitor::BinaryOperationVisitor;
106 using BinaryOperationVisitor::operator();
107
108 template<typename FloatType>
109 void operator()(std::span<FloatType> lhs, std::span<FloatType> rhs) {
110 using clean_t = std::decay_t<FloatType>;
111 auto pthis = this->make_this_eigen_tensor_<clean_t>();
112 auto plhs = this->make_lhs_eigen_tensor_(lhs);
113 auto prhs = this->make_rhs_eigen_tensor_(rhs);
114
115 pthis->addition_assignment(this_labels(), lhs_labels(), rhs_labels(),
116 *plhs, *prhs);
117 }
118};
