61  label_type lhs_permutation() const {
62  using split_string_type = typename label_type::split_string_type;
63  split_string_type rv;
64  auto lfree = lhs_free();
65  auto ldummy = lhs_dummy();
66  for(const auto& freei : m_result_) {
67  if(!lfree.count(freei)) continue;
68  rv.push_back(freei);
69  }
70  for(const auto& dummyi : ldummy) rv.push_back(dummyi);
71  return label_type(std::move(rv));
72  }
73 
75  label_type rhs_permutation() const {
76  typename label_type::split_string_type rv;
77  auto rfree = rhs_free();
78  auto rdummy = lhs_dummy(); // Use LHS dummy to get the same order!
79  for(const auto& dummyi : rdummy)
80  rv.push_back(dummyi); // Know it only appears 1x
81  for(const auto& freei : m_result_) {
82  if(!rfree.count(freei)) continue;
83  rv.push_back(freei); // Know it only appears 1x
84  }
85  return label_type(std::move(rv));
86  }
87 
96  label_type result_matrix_labels() const {
97  const auto lhs = lhs_permutation();
98  const auto rhs = rhs_permutation();
99  return lhs.concatenation(rhs).difference(lhs_dummy());
100  }
101 
102 private:
104  void assert_no_repeated_indices_() const {
105  const bool result_good = !m_result_.has_repeated_indices();
106  const bool lhs_good = !m_lhs_.has_repeated_indices();
107  const bool rhs_good = !m_rhs_.has_repeated_indices();
108 
109  if(result_good && lhs_good && rhs_good) return;
110  throw std::runtime_error("One or more terms contain repeated labels");
111  }
112 
114  void assert_dummy_indices_are_similar_() const {
115  if(lhs_dummy().is_permutation(rhs_dummy())) return;
116  throw std::runtime_error("Dummy indices must appear in all terms");
117  }
118 
120  void assert_no_shared_free_() const {
121  if(!lhs_free().intersection(rhs_free()).size()) return;
122  throw std::runtime_error("Contraction must sum repeated indices");
123  }
124 
125  label_type m_result_;

127  label_type m_rhs_;
